\begin{frame}
  \frametitle{Цифровые водяные знаки: общая информация}

  \begin{mydefinition}
    Цифровые водяные знаки (ЦВЗ) --- технология внедрения некоторой информации в цифровые данные.
  \end{mydefinition}

  \pause
  \begin{remark}
  Будем считать, что цель применения ЦВЗ --- защита от копирования. 
  \end{remark}

  \pause
  \begin{remark}
  В дальнейшем мы будем говорить о невидимых ЦВЗ.
  \end{remark}

  \pause
  Жизненный цикл ЦВЗ состоит из:
  \begin{itemize}
    \pause
    \item модификации исходных данных (внедрения ЦВЗ);
    \pause
    \item атаки модифицированных данных злоумышленником;
    \pause
    \item извлечения ЦВЗ из атакованных данных.
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{ЦВЗ для двумерных векторных данных}

  \begin{itemize}
  \pause
  \item Исходные данные --- планарный прямолинейный граф.
  \pause
  \begin{remark}
    Вершины заданы в какой-то координатной системе.
  \end{remark}

  \pause
  \item Внедрение ЦВЗ --- изменение координат вершин.

  \pause
  \item Виды атак:
  \begin{itemize}
    \pause \item переупорядочивание входных данных;
    \pause \item преобразование подобия на множестве вершин;
    \pause \item вставка и удаление вершин (упрощение и сглаживание конутров и полилиний);
    \pause \item применение аддитивного случайного шума к координатам вершин.
    \pause \item обрезка (cropping).
  \end{itemize}

  \pause 
  \item Требования к внедрению ЦВЗ:
  \begin{itemize}
    \item устойчивость к атакам;
    \item незначительное искажение исходных данных.
  \end{itemize}

  \end{itemize}    
\end{frame}

\begin{frame}
  \frametitle{Классификация алгоритмов внедрения ЦВЗ}

  \begin{itemize}
  \pause
  \item Использование исходных данных для извлечения ЦВЗ 
    \begin{itemize}
      \pause \item ``хорошо информированные'' (well-informed);
      \pause \item ``ослепленные'' (blinded).
    \end{itemize}
  
  \pause
  \item Область работы алгоритма
    \begin{itemize}
      \pause \item изменяются непосредственно координаты вершин (пространстенная область);
      \pause \item изменяются некоторые взаимно-однозначно связанные с координатами вершин коеффициенты (частотная область).
    \end{itemize}
  \end{itemize}

  \pause
  \begin{remark}
    Хорошо информированные частостные алгоритмы наиболее устойчивы к атакам. Лучший из известных автору --- алгоритм Обучи (Ohbuchi).
  \end{remark}
\end{frame}

\begin{frame}
  \frametitle{Схема работы алгоритма Обучи}

  \begin{itemize}
    \pause \item Разбиение плоскости на прямоугольные домены.
    \pause
    \begin{remark}
      Дальнейшие шаги производятся независимо для каждого домена.
    \end{remark}
    \pause \item Построение триангуляции Делоне.
    \pause \item Вычисление частотных коэффициентов --- разложение вектора составленного из координат вершин, в базисе собственных векторов графа триангуляции.
    \pause
    \begin{remark}
      СВ графа суть СВ невзвешенного лапласиана $R = I - D^{-1} A$.
    \end{remark}
    \pause \item Изменение частотных коэффициентов сообразно внедряемому сообщению.
    \pause \item Вычисление новых координат вершин по измененным частотным коэффициентам.
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Оценка искажения, непрерывный случай (1)}

  $\Omega \subset \mathbb{R}^2$ \pause , $f: \Omega \to \mathbb{R}^2$ \pause, 
  $f = \begin{pmatrix} f_x(x, y) \\ f_y(x, y) \end{pmatrix}$, 
  $\nabla f_x = \begin{pmatrix} \frac{\partial f_x}{\partial x} \\ \frac{\partial f_x}{\partial y} \end{pmatrix}, 
  \nabla f_y = \begin{pmatrix} \frac{\partial f_y}{\partial x} \\ \frac{\partial f_y}{\partial y} \end{pmatrix}$,
  $\nabla f = \begin{pmatrix} 
    \frac{\partial f_x}{\partial x} & \frac{\partial f_x}{\partial y} \\
    \frac{\partial f_y}{\partial x} & \frac{\partial f_y}{\partial y} \\
  \end{pmatrix}.$ 
\pause

\begin{frame}
  \frametitle{Оценка искажения, непрерывный случай (2)}
  \pause
  \begin{statement}
  $
  \left(\frac{\left| {q(r, dr, \pm \frac{\pi}{2})} \right|}{|dr|}\right)^2 = 
  \left(\frac{\partial f_x}{\partial y} + \frac{\partial f_y}{\partial x}\right)^2 + 
  \left(\frac{\partial f_x}{\partial x} - \frac{\partial f_y}{\partial y}\right)^2 
  $
  \end{statement}
  \pause
  \begin{statement}
  Функционал
   $E_C(f) = \frac{1}{2} \int_{\Omega}\left[{ \left(\frac{\partial f_x}{\partial x} - \frac{\partial f_y}{\partial y}\right)^2 + 
                                            \left(\frac{\partial f_x}{\partial y} + \frac{\partial f_y}{\partial x}\right)^2} \right] d\Omega$,
  называемый \textit{конформной энергией}, есть мера искажения функцией $f$ углов в области $\Omega$. 
  \end{statement}

  \pause
  \begin{remark}
    $$E_D(f) = E_C(f) - \int_{\Omega} \det (\nabla f) d\Omega = 
    \frac{1}{2} \int_{\Omega} \left[ \left| \nabla f_x \right| ^ 2 + \left| \nabla f_y \right| ^ 2 \right] d\Omega$$ называется
    \textit{энергия Дирихле}.
  \end{remark}
\end{frame}

\begin{frame}
  \frametitle{Оценка искажения, дискретный случай (1)}

  \pause
  \begin{statement}
  Пусть даны:
  \begin{itemize}
    \pause \item Множества $V = \left\{v_1, v_2, ... , v_n\right\}$, $V' = \left\{v_1', v_2', ... , v_n'\right\}$, $v_i, v_i' \in \mathbb{R}^2$;
    \pause \item $f: V \to V'$, $f(v_i) = v_i', 1 \le i \le n$; 
    \pause \item Триангуляция $\mathfrak{T}$ множества $V$, $\Omega$ -- выпуклая оболочка $V$;
  \end{itemize}
  \pause тогда определена кусочно-линейная интерполяционная поверхность $f: \Omega \to \mathbb{R}^2$. 
  \end{statement}

\pause
Пусть $G$ --- граф триангуляции $\mathfrak{T}$, $w_{uv} = \cot \alpha_{uv} + \cot \alpha_{vu}$.
\pause
\begin{statement}
\begin{equation*}
  L_{uv} = \begin{cases}
  \sum_{(v, x) \in E(G)}{w_{vx}}&\text{if $u = v,$} \\
  -w_{uv}&\text{if $u \sim v,$} \\
  0 &\text{otherwise;}
\end{cases} 
  E_D(f) = \frac{1}{4} \conjugate{\mathbf{f}^T} L \mathbf{f}.  
\end{equation*}
\end{statement}

\end{frame}

\begin{frame}
  \frametitle{Оценка искажения, дискретный случай (2)}

  \pause
  $L_{uv}' = \begin{cases}
    i  & \text{if edge $uv \in \partial E$ and $\Omega$ is left of the line $(u, v)$;} \\ 
    -i & \text{if edge $uv \in \partial E$ and $\Omega$ is right of the line $(u, v)$;} \\ 
    0  & \text{otherwise.}
  \end{cases}$ 
  \pause
  \begin{statement}
  $$\int_{\Omega} \det {\left( \nabla f \right) } d\Omega = \frac{1}{4} \conjugate{\mathbf{f}^T} L' \mathbf{f}, \mbox{   }L'' = L - L', \mbox{   } E_C(f) = \frac{1}{4} \conjugate{\mathbf{f}^T} L'' \mathbf{f}. $$
  \end{statement}
  \pause
  \begin{statement}
    Для минимизации коформной энергии вектор изменений координат должен быть линейной комбинацией младших собственных векторов эрмитовой матрицы $L''$.
  \end{statement}
\end{frame}
