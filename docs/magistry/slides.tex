\begin{frame}
  \frametitle{Постановка задачи}

    \begin{InputData}
    ГИС-данные --- геометрический планарный прямолинейный граф $G$, обладающий следующими свойствами:
    \begin{itemize}
        \pause \item кластеризованность;
        \pause \item равномерное распределение вершин внутри кластера;
        \pause \item характерный масштаб и допустимая погрешность.
    \end{itemize}
    \pause
    \end{InputData}

    \pause
    \begin{ProblemDef}
    \begin{itemize}
        \item Защитить авторские права на ГИС-данные, изменяя координаты вершин в пределах допустимой погрешности. 

        Изменение координат вершин не должно нарушать топологии графа.
        \pause
        \item Сформулировать критерий $K$ степени искажения исходных данных.
    \end{itemize}
    \end{ProblemDef}

\end{frame}

\begin{frame}
  \frametitle{Формальная постановка задачи}

    Найти $f(V_G, key, msg) \to V_G'$, такую что $\exists f^{-1}(g(V_G'), V_G, key) \to msg$, где
    \begin{itemize}
        \item $key$ --- секретный ключ;
        \item $msg$ --- внедряемое сообщение, удостоверяющее авторство;
        \item $g: V_G' \to V_G''$ --- атака злоумышленником.
    \end{itemize}
    Определить и минимизировать $K(G, V_G') \to \mathbb{R}$.

    \pause
    \begin{remark}
        Функция $f$ называется цифровым водяным знаком (ЦВЗ). 
        
        Применение $f$ называется внедрением ЦВЗ.
    \end{remark}

    \pause
    Атаки бывают четырех видов:
    \begin{itemize}
        \pause \item переупорядочивание входных данных;
        \pause \item преобразование подобия на множестве вершин;
        \pause \item вставка и удаление вершин (упрощение и сглаживание конутров и полилиний);
        \pause \item применение аддитивного шума к координатам вершин.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Алгоритмы внедрения ЦВЗ}

    Алгоритмы внедрения ЦВЗ условно делятся на
    \begin{itemize}
        \item изменяющие непосредственно координаты вершин (пространственная область), для этих алгоритмов легко контролировать сохранение топологии графа;
        \item изменяющие некоторые взаимно-однозначно связанные с координатами вершин коэффициенты (частотная область), эти алгоритмы более устойчивы к атакам.
    \end{itemize}
    \pause

    \pause
    Одним из вариантов сохранения топологии графа алгоритмами, работающими в частотной области, 
    является ограничение допустимых смещений вершин по обобщенной диаграмме Вороного. 
    \begin{figure}
       \includegraphics[scale=0.3]{huber.png}
    \end{figure}
    \pause

    Из известных автору наиболее устойчив к атакам алгоритм Обучи.

\end{frame}

\begin{frame}
  \frametitle{Схема работы алгоритма Обучи}

  \begin{itemize}
    \pause \item Построение триангуляции Делоне множества $V_G$.
    \pause \item Вычисление частотных коэффициентов --- разложение вектора, составленного из координат вершин, в базисе собственных векторов невзвешенного лапласиана графа триангуляции.
    %СВ графа суть СВ невзвешенного лапласиана графа. %$R = I - D^{-1} A$, где $A$~---~матрица смежности, $D$ --- диагональная матрица из степеней вершин.
    \pause \item Модуляция частотных коэффициентов внедряемым сообщениeм с использованием секретного ключа.
    \pause \item Вычисление новых координат вершин по модулированным частотным коэффициентам.
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Критерий степени искажения исходных данных}
    \begin{hyphotesis}
    Среднее изменение углов между смежными ребрами графа $G$ есть разумный критерий степени искажения исходных данных.
    \end{hyphotesis}

    \pause
    \begin{statement}
    Пусть дана дифференцируемая функция $f: (\Omega \subset \mathbb{R}^2) \to \mathbb{R}^2$, тогда функционал  
    $ E_C(f) = \frac{1}{2} \int_{\Omega}\left[{ \left(\frac{\partial f_x}{\partial x} - \frac{\partial f_y}{\partial y}\right)^2 + 
                                                \left(\frac{\partial f_x}{\partial y} + \frac{\partial f_y}{\partial x}\right)^2} \right] d\Omega$ 
    измеряет среднее изменение локальных углов под действием $f$. 
    \end{statement}

    \pause
    \begin{statement}
    Пусть дана функция $f: V_G \to \mathbb{R}^2$, $f_{\mathfrak{T}}: ConvexHull(V_G) \to \mathbb{R}^2$ есть кусочно-линейная интерполяция $f$ 
    по триангуляции $\mathfrak{T}$ множества $V_G$, тогда функционал $E_C(f_{\mathfrak{T}})$ измеряет среднее изменение углов $\mathfrak{T}$.
    \end{statement}
\end{frame}


\begin{frame}
  \frametitle{Минимизация искажения исходных данных}

  \begin{statement}
  Пусть $L$ --- взвешенный лапласиан графа триангуляции $\mathfrak{T}$, тогда $\textstyle E_C(f_{\mathfrak{T}}) = \conjugate{\mathbf{f}^T} L \mathbf{f}.$ 
  \end{statement}

  \pause
  \begin{statement}
    Для минимизации $E_C(f_{\mathfrak{T}})$ в алгоритме Обучи необходимо использовать взвешенный лапласиан $L$ и частотные коэффициенты, соответствующие 
    младшим собственным векторам $L$, при модуляции.

    \pause
    \vspace{0.5cm}

    Для минимизации предложенного критерия степени искажения исходных данных, необходимо выбрать $\mathfrak{T}$ как триангуляцию Делоне 
    множества $V(G)$ с ребрами $E(G)$ в качестве ограничений.
  \end{statement}
\end{frame}
